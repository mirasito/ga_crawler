**Шаг 2 — извлечь sample JSON-LD из product HTML:**

Запустить через Python:

```python
from pathlib import Path
import re, json
from selectolax.parser import HTMLParser

html_file = Path(".planning/spikes/01-goldapple/sample-payloads/goldapple-product-html-1.html")
html = html_file.read_text(encoding="utf-8")
tree = HTMLParser(html)

jsonld_blocks = []
for s in tree.css('script[type="application/ld+json"]'):
    raw = s.text()
    try:
        obj = json.loads(raw)
        jsonld_blocks.append(obj)
    except json.JSONDecodeError as e:
        print(f"Bad JSON-LD: {e}")

if jsonld_blocks:
    out = Path(".planning/spikes/01-goldapple/sample-payloads/goldapple-jsonld-sample.json")
    out.write_text(json.dumps(jsonld_blocks, indent=2, ensure_ascii=False), encoding="utf-8")
    print(f"Saved {len(jsonld_blocks)} JSON-LD blocks")
    # Print Product type if present
    for obj in jsonld_blocks:
        if isinstance(obj, dict) and obj.get("@type") == "Product":
            print(f"  Product: name={obj.get('name')}, price={obj.get('offers', {}).get('price')}")
else:
    print("No JSON-LD found in HTML — D-14 success criterion needs revisiting")
    Path(".planning/spikes/01-goldapple/sample-payloads/goldapple-jsonld-sample.json").write_text("[]", encoding="utf-8")
```
