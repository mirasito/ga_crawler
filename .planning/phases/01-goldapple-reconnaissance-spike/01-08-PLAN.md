```python
"""
Spike 01: Goldapple Patchright Tier 2 fetch experiment.

Reproducible: запускается из repo root через
  uv run python .planning/spikes/01-goldapple/notebook.py --mode kz-laptop
  uv run python .planning/spikes/01-goldapple/notebook.py --mode eu-proxy
  uv run python .planning/spikes/01-goldapple/notebook.py --mode tier3

Per CONTEXT.md decisions:
  D-01: Tier 2 (Patchright). Vanilla Playwright skipped.
  D-03: Stop-rule = 5 consecutive blocks OR first Cloudflare/DataDome challenge без auto-resolve.
  D-04: persistent context (warm), slow rate (3-5s pause), cookies live across fetches.
  D-13: Threshold ≥95/100 successful fetches (5xx/timeout retries allowed).
  D-14: Successful fetch = HTTP 200 + JSON-LD product schema present.
  D-15: Auto-resolved challenge = pass; challenge-rate logged separately. >20% = fragile.

Output:
  sample-payloads/tier2-{mode}-results.json
  sample-payloads/tier2-{mode}-log.txt
"""

import argparse
import json
import os
import random
import time
from datetime import datetime, timezone
from pathlib import Path

import structlog
from dotenv import load_dotenv
from patchright.sync_api import sync_playwright
from selectolax.parser import HTMLParser

# Config
URLS_FILE = Path(".planning/spikes/01-goldapple/sample-payloads/goldapple-product-urls.txt")
OUTPUT_DIR = Path(".planning/spikes/01-goldapple/sample-payloads")
USER_DATA_DIR = Path(".planning/spikes/01-goldapple/browser-state")
PAUSE_RANGE = (3.0, 5.0)        # D-04 random uniform
PAGE_TIMEOUT_MS = 60000
CHALLENGE_WAIT_S = 10           # доп. ожидание после первой загрузки если challenge
CONSECUTIVE_BLOCK_LIMIT = 5     # D-03 stop-rule
RETRY_PER_URL = 1               # D-13: 5xx/timeout retries разрешены

CHALLENGE_MARKERS = [
    "just a moment",
    "cf-mitigated",
    "cf-browser-verification",
    "cf-challenge",
    "datadome",
    "ddg.captcha",
    "checking your browser",
]

structlog.configure(
    processors=[
        structlog.processors.add_log_level,
        structlog.processors.TimeStamper(fmt="iso"),
        structlog.processors.JSONRenderer(),
    ]
)
log = structlog.get_logger()

def is_challenge_html(html: str) -> bool:
    lower = html.lower()
    return any(m in lower for m in CHALLENGE_MARKERS)

def has_jsonld_product(html: str) -> tuple[bool, bool]:
    """Returns (has_product, has_price). D-14 success criterion."""
    tree = HTMLParser(html)
    has_product = False
    has_price = False
    for s in tree.css('script[type="application/ld+json"]'):
        try:
            obj = json.loads(s.text())
        except (json.JSONDecodeError, AttributeError):
            continue
        items = obj if isinstance(obj, list) else [obj]
        # Handle @graph wrapper тоже
        flat = []
        for it in items:
            if isinstance(it, dict) and "@graph" in it:
                flat.extend(it["@graph"])
            else:
                flat.append(it)
        for it in flat:
            if not isinstance(it, dict):
                continue
            if it.get("@type") == "Product" or (isinstance(it.get("@type"), list) and "Product" in it["@type"]):
                has_product = True
                offers = it.get("offers", {})
                if isinstance(offers, dict) and offers.get("price"):
                    has_price = True
                elif isinstance(offers, list) and any(isinstance(o, dict) and o.get("price") for o in offers):
                    has_price = True
    return has_product, has_price

def fetch_one(page, url: str) -> dict:
    started = time.perf_counter()
    rec = {
        "url": url,
        "fetched_at": datetime.now(timezone.utc).isoformat(),
        "status": None,
        "timing_ms": None,
        "html_length": None,
        "jsonld_has_product": False,
        "jsonld_has_price": False,
        "challenge_detected": False,
        "challenge_resolved": False,
        "block": False,
        "block_reason": None,
        "error": None,
    }
    try:
        response = page.goto(url, wait_until="domcontentloaded", timeout=PAGE_TIMEOUT_MS)
        rec["status"] = response.status if response else None
        html = page.content()
        rec["html_length"] = len(html)

        if is_challenge_html(html):
            rec["challenge_detected"] = True
            time.sleep(CHALLENGE_WAIT_S)
            try:
                page.reload(wait_until="domcontentloaded", timeout=PAGE_TIMEOUT_MS)
            except Exception:
                pass
            html = page.content()
            if is_challenge_html(html):
                rec["block"] = True
                rec["block_reason"] = "challenge_not_resolved"
            else:
                rec["challenge_resolved"] = True

        if not rec["block"]:
            has_p, has_pr = has_jsonld_product(html)
            rec["jsonld_has_product"] = has_p
            rec["jsonld_has_price"] = has_pr
            if rec["status"] != 200:
                rec["block"] = True
                rec["block_reason"] = f"http_{rec['status']}"
            elif not has_p:
                rec["block"] = True
                rec["block_reason"] = "no_jsonld_product"
    except Exception as e:
        rec["error"] = f"{type(e).__name__}: {e}"
        rec["block"] = True
        rec["block_reason"] = "exception"
    rec["timing_ms"] = int((time.perf_counter() - started) * 1000)
    return rec

def make_proxy_settings(mode: str):
    if mode in ("eu-proxy", "tier3"):
        load_dotenv(".env.local")
        host = os.environ.get("PROXY_HOST")
        port = os.environ.get("PROXY_PORT")
        user = os.environ.get("PROXY_USER")
        password = os.environ.get("PROXY_PASS")
        if not all([host, port, user, password]):
            raise RuntimeError(f"Mode {mode} requires PROXY_* in .env.local — see plan 01-03")
        return {
            "server": f"http://{host}:{port}",
            "username": user,
            "password": password,
        }
    return None

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--mode", choices=["kz-laptop", "eu-proxy", "tier3"], default="kz-laptop")
    ap.add_argument("--limit", type=int, default=100)
    args = ap.parse_args()

    urls = [u.strip() for u in URLS_FILE.read_text(encoding="utf-8").splitlines() if u.strip()][:args.limit]
    print(f"Mode: {args.mode}, URLs: {len(urls)}, pause range: {PAUSE_RANGE}s")

    proxy = make_proxy_settings(args.mode)
    if proxy:
        print(f"Using proxy server: {proxy['server']} (creds redacted)")

    USER_DATA_DIR.mkdir(parents=True, exist_ok=True)
    log_file = OUTPUT_DIR / f"tier2-{args.mode}-log.txt"
    log_handle = log_file.open("w", encoding="utf-8")

    results = []
    consecutive_blocks = 0
    stopped_early = False
    stop_reason = None

    with sync_playwright() as p:
        kwargs = dict(
            headless=False,                                  # operator наблюдает за browser
            user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/130.0.0.0 Safari/537.36",
            viewport={"width": 1366, "height": 768},
            locale="ru-KZ",
        )
        if proxy:
            kwargs["proxy"] = proxy
        context = p.chromium.launch_persistent_context(
            user_data_dir=str(USER_DATA_DIR),
            **kwargs,
        )
        page = context.new_page()

        for i, url in enumerate(urls, 1):
            print(f"\n[{i}/{len(urls)}] {url}")
            attempts = 0
            rec = None
            while attempts <= RETRY_PER_URL:
                attempts += 1
                rec = fetch_one(page, url)
                if rec["block"] and rec["block_reason"] == "exception" and attempts <= RETRY_PER_URL:
                    print(f"  Retry {attempts} after error: {rec['error']}")
                    time.sleep(random.uniform(*PAUSE_RANGE))
                    continue
                break
            results.append(rec)
            log_handle.write(json.dumps(rec, ensure_ascii=False) + "\n")
            log_handle.flush()
            print(f"  -> status={rec['status']}, timing={rec['timing_ms']}ms, "
                  f"jsonld_product={rec['jsonld_has_product']}, "
                  f"challenge_detected={rec['challenge_detected']}, "
                  f"challenge_resolved={rec['challenge_resolved']}, "
                  f"block={rec['block']} ({rec['block_reason'] or 'OK'})")

            # D-03 stop-rule
            if rec["block"]:
                consecutive_blocks += 1
                if consecutive_blocks >= CONSECUTIVE_BLOCK_LIMIT:
                    stopped_early = True
                    stop_reason = f"{consecutive_blocks} consecutive blocks"
                    print(f"\n!!! STOP-RULE TRIGGERED: {stop_reason}")
                    break
                if rec["block_reason"] == "challenge_not_resolved":
                    stopped_early = True
                    stop_reason = f"Cloudflare/DataDome challenge not auto-resolved at fetch {i}"
                    print(f"\n!!! STOP-RULE TRIGGERED: {stop_reason}")
                    break
            else:
                consecutive_blocks = 0

            if i < len(urls):
                time.sleep(random.uniform(*PAUSE_RANGE))

        context.close()

    log_handle.close()

    success = sum(1 for r in results if not r["block"] and r["jsonld_has_product"])
    challenges = sum(1 for r in results if r["challenge_detected"])
    challenges_resolved = sum(1 for r in results if r["challenge_resolved"])
    blocks = sum(1 for r in results if r["block"])
    challenge_rate_pct = (challenges / len(results) * 100) if results else 0.0

    summary = {
        "mode": args.mode,
        "proxy_used": proxy is not None,
        "total_attempted": len(results),
        "total_planned": len(urls),
        "success_count": success,
        "challenge_count": challenges,
        "challenge_resolved_count": challenges_resolved,
        "block_count": blocks,
        "challenge_rate_pct": round(challenge_rate_pct, 1),
        "stopped_early": stopped_early,
        "stop_reason": stop_reason,
        "passes_d13_threshold": success >= 95,                 # D-13
        "fragile_per_d15": challenge_rate_pct > 20,            # D-15
    }
    out = {"summary": summary, "results": results}
    out_path = OUTPUT_DIR / f"tier2-{args.mode}-results.json"
    out_path.write_text(json.dumps(out, indent=2, ensure_ascii=False), encoding="utf-8")

    print(f"\n{'='*60}")
    print(f"SUMMARY: success={success}/{len(results)}, "
          f"challenges={challenges} ({challenge_rate_pct:.1f}%), "
          f"resolved={challenges_resolved}, blocks={blocks}, "
          f"stopped_early={stopped_early}")
    print(f"D-13 (>=95/100): {'PASS' if summary['passes_d13_threshold'] else 'FAIL'}")
```
