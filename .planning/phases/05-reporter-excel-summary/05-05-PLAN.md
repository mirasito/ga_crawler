def _setup_successful_run(tmp_path: Path) -> tuple[Path, int]:
    """Plant a successful run with paired snapshots and matches; return (db_path, run_id)."""
    from sqlalchemy import text as _text

    from ga_crawler.storage.sqlite import (
        SqliteRunWriter, SqliteSnapshotWriter, init_db, make_engine,
    )

    db_path = tmp_path / "prices.db"
    init_db(db_path)
    engine = make_engine(db_path)
    run_writer = SqliteRunWriter(engine)
    snap_writer = SqliteSnapshotWriter(engine)

    run_id = run_writer.create(
        started_at=datetime(2026, 5, 10, 14, tzinfo=timezone.utc)
    )

    base_kwargs = dict(
        volume_raw="50 мл", currency="KZT", stock_state="IN_STOCK",
        brand_norm="givenchy", name_norm="edp", volume_norm="(50, ml, 1)",
        multipack_flag=0,
    )
    snap_writer.append(run_id, "viled", [
        dict(sku_id="v1", url="https://viled.kz/p/1", name="X", brand="Givenchy",
             current_price=50000, was_price=None,
             scraped_at=datetime(2026, 5, 10, 14, 5, tzinfo=timezone.utc),
             **base_kwargs)
    ])
    snap_writer.append(run_id, "goldapple", [
        dict(sku_id="g1", url="https://goldapple.kz/p/1", name="X", brand="Givenchy",
             current_price=55000, was_price=None,
             scraped_at=datetime(2026, 5, 10, 14, 10, tzinfo=timezone.utc),
             **base_kwargs)
    ])

    # Plant one match row
    with engine.begin() as conn:
        conn.execute(_text(
            "INSERT INTO matches (run_id, viled_sku, goldapple_sku, brand_norm, "
            "name_norm, volume_norm, viled_price, goldapple_price, viled_was_price, "
            "goldapple_was_price, price_delta, price_delta_pct, matched_at) "
            "VALUES (:rid, 'v1', 'g1', 'givenchy', 'edp', '(50, ml, 1)', "
            "50000, 55000, NULL, NULL, 5000, 10.0, '2026-05-10T14:20:00Z')"
        ), {"rid": run_id})

    # Pre-populate runs.stats with upstream namespaces + finalize to success
    run_writer.patch_stats(run_id, {
        "viled.fetch_count": 1, "goldapple.fetch_count": 1,
        "match.count": 1, "match.rate": 100.0,
    })
    run_writer.finalize(run_id, status="success")

    return db_path, run_id


def _write_pyproject(tmp_path: Path) -> Path:
    pyp = tmp_path / "pyproject.toml"
    pyp.write_text(
        "[tool.ga_crawler.report]\n"
        "output_dir = \"reports\"\n"
        "size_limit_mb = 45\n"
        "top_n_deltas = 3\n"
        "timezone = \"Asia/Almaty\"\n",
        encoding="utf-8",
    )
    return pyp


def _run_cli(args: list, cwd: Path) -> subprocess.CompletedProcess:
    """Invoke `python -m ga_crawler` as subprocess, capture stdout+stderr."""
    cmd = [sys.executable, "-m", "ga_crawler"] + args
    return subprocess.run(
        cmd, capture_output=True, text=True, cwd=str(cwd), encoding="utf-8"
    )


def test_report_run_missing_run_id_exits_2(tmp_path):
    result = _run_cli(["report-run"], cwd=tmp_path)
    assert result.returncode == 2  # argparse error
    assert "run-id" in (result.stderr or "")


def test_report_run_non_int_run_id_exits_2(tmp_path):
    result = _run_cli(["report-run", "--run-id", "abc"], cwd=tmp_path)
    assert result.returncode == 2
    assert "invalid" in (result.stderr or "").lower() or "int" in (result.stderr or "").lower()


def test_report_run_nonexistent_run_exits_2(tmp_path):
    """Non-existent run_id → reporter skips, exit code 2."""
    db_path = tmp_path / "empty.db"
    pyp = _write_pyproject(tmp_path)
    # init the empty DB so SqliteRunWriter works
    from ga_crawler.storage.sqlite import init_db
    init_db(db_path)

    result = _run_cli(
        [
            "report-run", "--run-id", "99999",
            "--db-path", str(db_path),
            "--pyproject", str(pyp),
            "--repo-root", str(tmp_path),
        ],
        cwd=tmp_path,
    )
    assert result.returncode == 2
    payload = json.loads(result.stdout)
    assert payload["status"] == "skipped"
    assert payload["reason"] == "missing_run_row"


def test_report_run_success_writes_xlsx_and_exits_0(tmp_path):
    db_path, run_id = _setup_successful_run(tmp_path)
    pyp = _write_pyproject(tmp_path)

    result = _run_cli(
        [
            "report-run", "--run-id", str(run_id),
            "--db-path", str(db_path),
            "--pyproject", str(pyp),
            "--repo-root", str(tmp_path),
        ],
        cwd=tmp_path,
    )
    assert result.returncode == 0, f"stderr={result.stderr}\nstdout={result.stdout}"
    payload = json.loads(result.stdout)
    assert payload["status"] == "success"
    assert payload["run_id"] == run_id
    assert payload["xlsx_path"].endswith(".xlsx")
    assert payload["xlsx_size_bytes"] > 0
    assert payload["size_guard_passed"] is True

    # File exists on disk
    xlsx_abs = tmp_path / payload["xlsx_path"]
    assert xlsx_abs.exists()


def test_report_run_output_dir_override(tmp_path):
    db_path, run_id = _setup_successful_run(tmp_path)
    pyp = _write_pyproject(tmp_path)

    result = _run_cli(
        [
            "report-run", "--run-id", str(run_id),
            "--db-path", str(db_path),
            "--output-dir", "custom_reports",
            "--pyproject", str(pyp),
            "--repo-root", str(tmp_path),
        ],
        cwd=tmp_path,
    )
    assert result.returncode == 0, f"stderr={result.stderr}\nstdout={result.stdout}"
    payload = json.loads(result.stdout)
    assert payload["xlsx_path"].startswith("custom_reports/")
    assert (tmp_path / payload["xlsx_path"]).exists()


def test_report_run_idempotent(tmp_path):
    """Two consecutive invocations both exit 0; xlsx overwritten."""
    db_path, run_id = _setup_successful_run(tmp_path)
    pyp = _write_pyproject(tmp_path)
    args = [
        "report-run", "--run-id", str(run_id),
        "--db-path", str(db_path),
        "--pyproject", str(pyp),
        "--repo-root", str(tmp_path),
    ]
    r1 = _run_cli(args, cwd=tmp_path)
    assert r1.returncode == 0
    r2 = _run_cli(args, cwd=tmp_path)
    assert r2.returncode == 0

    p1 = json.loads(r1.stdout)
    p2 = json.loads(r2.stdout)
    assert p1["xlsx_path"] == p2["xlsx_path"]
    assert p1["status"] == p2["status"] == "success"
