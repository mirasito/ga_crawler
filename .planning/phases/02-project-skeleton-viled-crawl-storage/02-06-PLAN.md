```bash
#!/usr/bin/env bash
# Online SQLite backup + 4-rotate retention. D-219.
# Source: 02-RESEARCH.md §Don't Hand-Roll; 02-CONTEXT.md D-219.
# Phase 7 ops-playbook adds cron entry: "0 1 * * * /opt/ga_crawler/bin/backup.sh"
# (daily 01:00 KZ, after weekly run on Sunday → DB checkpointed by then).

set -euo pipefail

DB_PATH="${1:-prices.db}"
BACKUP_DIR="${2:-backups}"

if [ ! -f "$DB_PATH" ]; then
  echo "ERROR: source DB not found: $DB_PATH" >&2
  exit 1
fi

mkdir -p "$BACKUP_DIR"
TIMESTAMP=$(date +%Y-%m-%d)
TARGET="$BACKUP_DIR/$TIMESTAMP.db"

# Online .backup is atomic + WAL-safe (RESEARCH Pitfall 3 mitigation).
sqlite3 "$DB_PATH" ".backup '$TARGET'"

if [ ! -s "$TARGET" ]; then
  echo "ERROR: backup file empty or missing: $TARGET" >&2
  exit 2
fi

# Retention rotation: keep 4 most recent (DATA-06 minimum).
ls -t "$BACKUP_DIR"/*.db 2>/dev/null | tail -n +5 | xargs -r rm -f

echo "OK: backup written to $TARGET"
```

```python
"""DATA-06 backup script verification.

Source: 02-RESEARCH.md §Don't Hand-Roll Backup; 02-CONTEXT.md D-219.
"""
import os
import shutil
import sqlite3
import subprocess
import time
from datetime import date
from pathlib import Path

import pytest

from ga_crawler.storage.sqlite import init_db

BACKUP_SCRIPT = Path("bin/backup.sh").resolve()


def _platform_can_run_bash() -> bool:
    """Skip these tests on Windows where bash isn't available natively.
    On Windows, bash is available via Git for Windows / WSL — assume available
    if /usr/bin/bash or bash on PATH. Otherwise skip."""
    return shutil.which("bash") is not None


pytestmark = pytest.mark.skipif(
    not _platform_can_run_bash(),
    reason="bin/backup.sh requires bash on PATH (Git Bash on Windows OK)",
)


def _run_backup(db_path: Path, backup_dir: Path) -> subprocess.CompletedProcess:
    return subprocess.run(
        ["bash", str(BACKUP_SCRIPT), str(db_path), str(backup_dir)],
        capture_output=True, text=True,
    )


def test_backup_creates_valid_sqlite(tmp_path):
    db = tmp_path / "src.db"
    init_db(db)
    bdir = tmp_path / "backups"
    proc = _run_backup(db, bdir)
    assert proc.returncode == 0, f"stderr={proc.stderr}"
    today = date.today().isoformat()
    backup_path = bdir / f"{today}.db"
    assert backup_path.exists(), f"backup file missing: {backup_path}"
    # Verify it's a valid SQLite database
    conn = sqlite3.connect(str(backup_path))
    try:
        rows = conn.execute(
            "SELECT name FROM sqlite_master WHERE type='table'"
        ).fetchall()
        names = {r[0] for r in rows}
        assert "runs" in names
        assert "snapshots" in names
    finally:
        conn.close()


def test_backup_4_rotation_retention(tmp_path):
    db = tmp_path / "src.db"
    init_db(db)
    bdir = tmp_path / "backups"
    bdir.mkdir()
    # Create 6 stale .db files with descending mtimes
    now = time.time()
    for i in range(6):
        p = bdir / f"2025-01-{i+10:02d}.db"
        p.write_bytes(b"SQLite format 3\x00")  # minimal valid header
        # Set mtime so older = farther in past
        os.utime(p, (now - (i + 1) * 86400, now - (i + 1) * 86400))
    proc = _run_backup(db, bdir)
    assert proc.returncode == 0, f"stderr={proc.stderr}"
    remaining = sorted(bdir.glob("*.db"))
    # 6 stale + 1 new = 7; rotation keeps 4 most recent → 3 deleted
    assert len(remaining) == 4, f"expected 4 remaining, got {len(remaining)}: {[p.name for p in remaining]}"


def test_backup_fails_on_missing_source(tmp_path):
    bdir = tmp_path / "backups"
    proc = _run_backup(tmp_path / "nonexistent.db", bdir)
    assert proc.returncode != 0


def test_backup_creates_dir_if_missing(tmp_path):
    db = tmp_path / "src.db"
    init_db(db)
    bdir = tmp_path / "fresh_backups"  # does NOT exist
    assert not bdir.exists()
    proc = _run_backup(db, bdir)
    assert proc.returncode == 0
    assert bdir.exists()
    today = date.today().isoformat()
    assert (bdir / f"{today}.db").exists()
```
