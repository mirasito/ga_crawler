```python
"""viled __NEXT_DATA__ extractor.

PARSE-02 inversion: viled emits __NEXT_DATA__ for 15/15 PDPs and 0/15 JSON-LD per
spike 01-07. There is NO JSON-LD fallback path here; the only data source is the
inline `<script id="__NEXT_DATA__" type="application/json">` payload.

Source: 02-RESEARCH.md §Pattern 1 lines 376-446 (verbatim).
Wave-0-corrected paths: see .planning/phases/02-project-skeleton-viled-crawl-storage/02-WAVE0-PROBE.md
  - A1: in_stock / count / availability field name verified
  - A2: was_price discount semantics verified

Decisions: D-217 stock-state mapping; PARSE-04 sanity range [100, 1_000_000].
Pitfall 2: attributes[0] picks first variant — beauty SKUs typically have ≤1 size variant.
"""

from __future__ import annotations

import json
import re
from dataclasses import dataclass
from typing import Optional

import structlog
from selectolax.parser import HTMLParser

log = structlog.get_logger(__name__)

# Sanity range per PARSE-04 (mirror Phase 3 microdata `100 ≤ price ≤ 1_000_000`)
_PRICE_MIN = 100
_PRICE_MAX = 1_000_000

_SKU_FROM_URL_RE = re.compile(r"/item/(\d+)")


@dataclass(frozen=True)
class ViledRawProduct:
    """Raw extracted product from a viled PDP. Plan 05 normalizers consume this.

    9-field shape mirrors Phase 3 GoldappleRawProduct (parsers/goldapple_microdata.py).
    """
    sku_id: str
    url: str
    name: str
    brand_raw: str
    current_price: int
    was_price: Optional[int]
    currency: str
    availability: str  # StockState value
    raw_volume_text: Optional[str]


def _extract_next_data(html: str) -> Optional[dict]:
    """Extract `<script id="__NEXT_DATA__" type="application/json">` payload."""
    if not html:
        return None
    tree = HTMLParser(html)
    node = tree.css_first('script#__NEXT_DATA__')
    if node is None:
        return None
    raw = node.text(strip=False)
    if not raw:
        return None
    try:
        return json.loads(raw)
    except json.JSONDecodeError as e:
        log.warning("nextdata_parse_error", error=str(e))
        return None


def _map_stock_state(in_stock_raw: object) -> str:
    """D-217 mapping. Wave-0-corrected: see 02-WAVE0-PROBE.md §A1.

    Default mapping (boolean in_stock):
      True  → IN_STOCK
      False → OUT_OF_STOCK
      other → UNKNOWN

    If 02-WAVE0-PROBE.md verified an alternate field (e.g. `count > 0`),
    callers MUST adjust the in_stock_raw extraction in parse_pdp BEFORE
    invoking this function.
    """
    if in_stock_raw is True:
        return "IN_STOCK"
    if in_stock_raw is False:
        return "OUT_OF_STOCK"
    return "UNKNOWN"


def parse_pdp(html: str, url: str = "") -> Optional[ViledRawProduct]:
    """Extract product fields from viled __NEXT_DATA__.

    Returns None on:
      - missing __NEXT_DATA__ (likely 404 / gate)
      - JSON parse failure
      - empty attributes list (Pitfall 2 — no canonical price)
      - current_price outside sanity range (PARSE-04)
      - missing required keys

    Source: 02-RESEARCH.md §Pattern 1 lines 396-446.
    """
    nd = _extract_next_data(html)
    if nd is None:
        return None
    try:
        page_props = nd["props"]["pageProps"]
        item = page_props["item"]
        attrs = page_props.get("attributes") or []
        if not attrs:
            return None
        a0 = attrs[0]
        current_price_raw = a0.get("realPrice")
        if current_price_raw is None:
            return None
        try:
            current_price = int(current_price_raw)
        except (TypeError, ValueError):
            return None
        if not (_PRICE_MIN <= current_price <= _PRICE_MAX):
            return None  # PARSE-04 sanity range
        # was_price: original price IF > realPrice (discounted), else None
        full_price_raw = a0.get("price", current_price)
        try:
            full_price = int(full_price_raw)
        except (TypeError, ValueError):
            full_price = current_price
        was_price = full_price if full_price > current_price else None
        # currency: ₸ → KZT hardcoded (STATE.md plan 01-07 lock).
        # Unconditional KZT — viled is KZ-only retailer; any non-KZT raw value is
        # logged for observability but never propagates to the snapshot.
        currency_raw = (a0.get("currency") or "").strip()
        if currency_raw and currency_raw not in ("₸", "KZT"):
            log.warning("viled_unexpected_currency", currency_raw=currency_raw, url=url)
        currency = "KZT"
        # stock-state: per D-217 / Wave 0 PROBE corrected path
        # Default: in_stock boolean. If PROBE revised, swap field name HERE.
        in_stock_raw = a0.get("in_stock")
        availability = _map_stock_state(in_stock_raw)
        # sku_id: extract from URL `/item/(\d+)`, fall back to last path segment
        sku_match = _SKU_FROM_URL_RE.search(url)
        if sku_match:
            sku_id = sku_match.group(1)
        else:
            sku_id = url.rstrip("/").rsplit("/", 1)[-1]
        name = item.get("name") or ""
        brand_raw = item.get("brandName") or ""
        if not name or not brand_raw:
            return None
        return ViledRawProduct(
            sku_id=sku_id,
            url=url,
            name=name,
            brand_raw=brand_raw,
            current_price=current_price,
            was_price=was_price,
            currency=currency,
            availability=availability,
            raw_volume_text=name,  # NORM-03 extracts volume regex from full name
        )
    except (KeyError, TypeError, ValueError) as e:
        log.warning("viled_parse_error", url=url, error=str(e), error_type=type(e).__name__)
        return None


__all__ = ["ViledRawProduct", "parse_pdp", "_extract_next_data"]
```

```python
"""Per-retailer parser dispatcher — concrete ParseDispatcherProtocol impl.

Phase 3 registers `goldapple_microdata.parse_pdp` (FROZEN);
Phase 2 registers `viled_nextdata.parse_pdp` (NEW).

Source: 02-PATTERNS.md lines 631-666; 02-CONTEXT.md D-213 per-retailer split.
"""

from __future__ import annotations

from dataclasses import asdict, is_dataclass
from typing import Callable, Optional

from ga_crawler.parsers.goldapple_microdata import parse_pdp as parse_goldapple
from ga_crawler.parsers.viled_nextdata import parse_pdp as parse_viled


class ParseDispatcher:
    """Routes raw HTML to the right per-retailer parser.

    Returns the parsed product as a dict (so the orchestrator can pass it through
    NormalizerProtocol + SnapshotWriter without retailer-aware unwrapping).

    Conforms to interfaces.py ParseDispatcherProtocol (which expects
    `dispatch(retailer, html_or_data) -> Optional[dict]`). The `url` arg is
    forwarded to per-retailer parsers but optional in the Protocol; default to "".
    """

    _registry: dict[str, Callable[[str, str], object]] = {
        "viled": parse_viled,
        "goldapple": parse_goldapple,
    }

    def dispatch(self, retailer: str, html_or_data: str, url: str = "") -> Optional[dict]:
        parser = self._registry.get(retailer)
        if parser is None:
            return None
        result = parser(html_or_data, url)
        if result is None:
            return None
        if is_dataclass(result):
            return asdict(result)
        if isinstance(result, dict):
            return result
        return None


__all__ = ["ParseDispatcher"]
```

```python
"""Catalog/1310 enumeration via __NEXT_DATA__ pagination.

D-223: NOT sitemap-only; sitemap=42k luxury items without category metadata.
D-224: Mechanism = __NEXT_DATA__ pagination on 2 catalog endpoints (men + women).

Source: 02-RESEARCH.md §Pattern 2 lines 448-479 (verbatim).
Wave-0-corrected pagination keys: see 02-WAVE0-PROBE.md §A4 (totalCount / pageSize / products vs alternate names).

Pitfall 8: if catalog endpoint returns 403 / login redirect, this module returns []
and orchestrator (Plan 05) falls back to sitemap+brand-allowlist documented in PROBE.
"""

from __future__ import annotations

import time
from typing import Optional

import structlog
from curl_cffi import requests
from tenacity import (
    retry,
    retry_if_exception_type,
    stop_after_attempt,
    wait_exponential_jitter,
)

from ga_crawler.parsers.viled_nextdata import _extract_next_data

log = structlog.get_logger(__name__)

CATALOG_TIMEOUT_S = 30
CATALOG_PAUSE_S = 2.0  # D-225 — also overridable via ViledConfig.pause_seconds


class TransientFetchError(RuntimeError):
    """Raised on retryable network failures (5xx, network error, timeout)."""


@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential_jitter(initial=2, max=30),
    retry=retry_if_exception_type((TransientFetchError,)),
    reraise=True,
)
def _fetch_html(url: str, timeout_s: int = CATALOG_TIMEOUT_S) -> tuple[int, str]:
    """Fetch HTML with curl_cffi impersonate=chrome.
    Returns (status, body). Raises TransientFetchError on 5xx / network error.
    """
    try:
        resp = requests.get(url, impersonate="chrome", timeout=timeout_s)
    except Exception as e:
        raise TransientFetchError(f"connection error fetching {url}: {e}") from e
    if 500 <= resp.status_code < 600:
        raise TransientFetchError(f"http {resp.status_code} for {url}")
    return resp.status_code, resp.text


def fetch_catalog_urls(
    catalog_base: str,
    *,
    pause_seconds: float = CATALOG_PAUSE_S,
    fetch_callable=None,  # injectable for testing (Pitfall 1 — respx incompat)
) -> list[str]:
    """Walk catalog/1310 page-1 → totalCount/pageSize → enumerate ?page=2..N.

    Returns flat list of product URLs (likely /item/{numeric_id} per spike).
    Returns [] on Pitfall 8 (catalog blocked) — caller falls back to sitemap.
    """
    if fetch_callable is None:
        fetch_callable = _fetch_html
    status, page1_html = fetch_callable(catalog_base)
    if status != 200:
        log.warning("catalog_fetch_non_200", url=catalog_base, status=status)
        return []
    nd = _extract_next_data(page1_html)
    if nd is None:
        log.warning("catalog_no_nextdata", url=catalog_base)
        return []
    page_props = nd.get("props", {}).get("pageProps", {})
    products = page_props.get("products") or []
    # Wave-0-corrected key names — defaults match RESEARCH §Pattern 2; PROBE may revise.
    total = int(page_props.get("totalCount") or len(products))
    per_page = int(page_props.get("pageSize") or len(products) or 1)
    urls: list[str] = []
    for p in products:
        url = p.get("url") if isinstance(p, dict) else None
        if url:
            urls.append(_canonicalize(url, catalog_base))
    if per_page <= 0:
        return urls
    num_pages = (total + per_page - 1) // per_page
    for page in range(2, num_pages + 1):
        time.sleep(pause_seconds)
        page_url = f"{catalog_base}?page={page}"
        try:
            status, page_html = fetch_callable(page_url)
        except TransientFetchError as e:
            log.error("catalog_page_failed", page=page, url=page_url, error=str(e))
            continue
        if status != 200:
            log.warning("catalog_page_non_200", page=page, status=status)
            continue
        page_nd = _extract_next_data(page_html)
        if page_nd is None:
            continue
        page_products = page_nd.get("props", {}).get("pageProps", {}).get("products") or []
        for p in page_products:
            url = p.get("url") if isinstance(p, dict) else None
            if url:
                urls.append(_canonicalize(url, catalog_base))
    log.info("catalog_enumeration_complete", base=catalog_base, total=total, urls_collected=len(urls))
    return urls


def _canonicalize(url: str, base: str) -> str:
    """Convert relative product URL to absolute viled.kz URL."""
    if url.startswith("http"):
        return url
    if url.startswith("/"):
        # https://viled.kz/men/catalog/1310 → https://viled.kz
        from urllib.parse import urlparse
        parsed = urlparse(base)
        return f"{parsed.scheme}://{parsed.netloc}{url}"
    return f"{base.rstrip('/')}/{url}"
```

```python
"""viled fetcher — curl_cffi Tier 0, sync sequential, 2s rate-limit, per-SKU isolated.

D-225: NOT async. Single-thread `for` loop with `time.sleep(2.0)` between fetches.
Spike 01-07 confirmed 15/15 success at 2s pause via curl_cffi impersonate=chrome.

Source: 02-RESEARCH.md §Pattern 9 lines 829-857; §Pattern: Per-SKU Isolation;
       02-PATTERNS.md lines 51-153 (analog).
Wave-0-verified curl_cffi exception class names: see 02-WAVE0-PROBE.md §A10.
"""

from __future__ import annotations

import time
from typing import Callable, Optional

import structlog
from curl_cffi import requests
from tenacity import (
    retry,
    retry_if_exception_type,
    stop_after_attempt,
    wait_exponential_jitter,
)

log = structlog.get_logger(__name__)

VILED_TIMEOUT_S = 30
VILED_PAUSE_S = 2.0  # D-225, configurable via ViledConfig


class TransientFetchError(RuntimeError):
    """Raised on retryable network failures (5xx, network error, timeout)."""


@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential_jitter(initial=2, max=30),
    retry=retry_if_exception_type((TransientFetchError,)),
    reraise=True,
)
def _fetch_html(url: str, timeout_s: int = VILED_TIMEOUT_S) -> tuple[int, str]:
    """Fetch HTML with curl_cffi impersonate=chrome.
    Returns (status, body). Raises TransientFetchError on 5xx / network error.
    4xx returns naturally (caller decides — typically map to DELISTED for 404).
    """
    try:
        resp = requests.get(url, impersonate="chrome", timeout=timeout_s)
    except Exception as e:
        raise TransientFetchError(f"connection error fetching {url}: {e}") from e
    if 500 <= resp.status_code < 600:
        raise TransientFetchError(f"http {resp.status_code} for {url}")
    return resp.status_code, resp.text


def fetch_one_isolated(
    fetch_callable: Callable[[str], dict],
    url: str,
    stats: dict,
) -> Optional[dict]:
    """CRAWL-03 per-SKU isolation. Exception → log + count, return None.

    Source: 02-PATTERNS.md §"Pattern: Per-SKU Isolation" (verbatim adapt; sync).
    """
    try:
        return fetch_callable(url)
    except Exception as e:
        log.error(
            "fetch_failed",
            url=url,
            error=str(e),
            error_type=type(e).__name__,
        )
        stats["fetch_failures"] = stats.get("fetch_failures", 0) + 1
        return None


class ViledFetcher:
    """Sync curl_cffi fetcher. concurrency=1 + 2s pause between fetches (D-225)."""

    site = "viled"

    def __init__(self, *, run_id: int, pause_seconds: float = VILED_PAUSE_S):
        self.run_id = run_id
        self.pause_seconds = pause_seconds

    def fetch_one(self, url: str) -> dict:
        """Fetch one URL; return dict with status + html.
        Raises TransientFetchError if all 3 retries fail (caller's responsibility
        to wrap with fetch_one_isolated for run-loop continuity)."""
        status, html = _fetch_html(url)
        return {"status": status, "url": url, "html": html}

    def run_loop(
        self,
        urls: list[str],
        stats: dict,
        sleep_fn: Callable[[float], None] = time.sleep,
    ) -> list[dict]:
        """Sequential fetch loop with 2s pacing (CRAWL-06 + D-225)."""
        records: list[dict] = []
        for i, url in enumerate(urls, 1):
            rec = fetch_one_isolated(self.fetch_one, url, stats)
            stats["fetch_count"] = stats.get("fetch_count", 0) + 1
            if rec is not None:
                records.append(rec)
            log.info("fetch_progress", run_id=self.run_id, idx=i, total=len(urls), url=url)
            if i < len(urls):
                sleep_fn(self.pause_seconds)
        return records


__all__ = ["ViledFetcher", "fetch_one_isolated", "_fetch_html", "TransientFetchError"]
```
